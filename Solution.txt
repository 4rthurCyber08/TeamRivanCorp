Solution

1. Configure Etherchannel on the links between CSW1 & CSW2
	-Use the standard protocol for link aggregation
	-CSW1 must actively negotiate forming an etherchannel
	-CSW2 must passively negotiate forming an etherchannel
	-Use the specified Port Channel number

@CSW1
conf t
 int range fa0/10-11
  channel-group 17 mode active
  channel-protocol lacp
  end

@CSW2
conf t
 int range fa0/10-11
  channel-group 17 mode passive
  channel-protocol lacp
  end
  
2. Configure Trunk links between CSW1, CSW2, ASW1, ASW2, & ASW3 
using the standard trunking protocol with the following settings.
	- Set the native VLAN to VLAN 'Main'
	- Allow only VLANs:
		Admin
		Security
		Voice
		Main
		Guest
		HR
		Printers

@CSW1 & CSW2
conf t
 int po17
  sw tru encaps dot1q
  sw tru allowed vlan 12,25,65,69,80,100
  sw tru native vlan 69
  sw mo tr
 int range fa0/1-3
  sw tru encaps dot1q
  sw tru allowed vlan 12,25,65,69,80,100
  sw tru native vlan 69
  sw mo tr
  end

@ASW1
conf t
 int range fa0/4-5
  !sw tru encaps dot1q
  sw tru allowed vlan 12,25,65,69,80,100
  sw tru native vlan 69
  sw mo tr
  end

@ASW2 & ASW3
conf t
 int range fa0/3-4
  !sw tru encaps dot1q
  sw tru allowed vlan 12,25,65,69,80,100
  sw tru native vlan 69
  sw mo tr
  end

3. Configure PVST.
	-Assign CSW1 as the primary root bridge
	-Assign CSW2 as the secondary bridge

@CSW1
conf t
 spanning-tree vlan 12,25,65,69,80,100,120 root primary
 end

@CSW2
conf t
 spanning-tree vlan 12,25,65,69,80,100,120 root secondary
 end
  
4. Configure DHCP for Admin and Voice VLANs
	-Configure a DHCP server on CSW1 for Admin VLAN
		-Exclude the first 120 ips for Admin VLAN
		-set the default gateway for Admin VLAN to CSW1
		-set the domain name to: siteACenter.com
	
	-Configure DHCP on CSW2 for Voice VLAN
		-Exclude the first 60 ips for Admin VLAN
		-set the default gateway for Admin VLAN to CSW2
		-set the domain name to: siteAVoip.com

@CSW1
conf t
 ip dhcp excluded-add 10.0.12.1  10.0.12.120
 !pool name not specified. Choose any.
 ip dhcp pool adminPool 
  network 10.0.12.0 255.255.255.0
  default-router 10.0.12.1
  domain-name siteACenter.com
  end

@CSW2
conf t
 ip dhcp excluded-add 10.0.65.1  10.0.65.60
 !pool name not specified. Choose any.
 ip dhcp pool voicePool 
  network 10.0.65.0 255.255.255.0
  default-router 10.0.65.254
  domain-name siteAVoip.com
  end

5. Assign end devices to their specified VLANs

@ASW1
conf t
 int range fa0/1-2
  sw mo ac
  sw ac vlan 12
 int fa0/2
  sw vo vlan 65
 int fa0/3
  sw mo ac
  sw ac vlan 120
  end

@ASW2
conf t
 int fa0/1
  sw mo ac
  sw ac vlan 25
  sw vo vlan 65
 int fa0/2
  sw mo ac
  sw ac vlan 120
  end

@ASW3
conf t
 int fa0/1
  sw mo ac
  sw ac vlan 100
 int fa0/2
  sw mo ac
  sw ac vlan 80
  end

6. Configuree OSPF on CSW1, CSW2, R1, R2, R3
	-Point-to-point
	-DROTHER

@CSW1
conf t
 ip routing
 router ospf 1
  router-id 10.10.10.10
  network 10.0.12.0 0.0.0.255 area 0
  network 10.0.65.0 0.0.0.255 area 0
  network 10.0.69.16 0.0.0.15 area 0
  network 10.0.80.0 0.0.0.255 area 0
  network 10.0.100.0 0.0.0.255 area 0
  network 192.168.25.0 0.0.0.255 area 0
  network 10.10.10.10 0.0.0.0 area 0
  network 10.2.4.0 0.0.0.3 area 0
  end

@CSW2
conf t
 ip routing
 router ospf 1
  router-id 20.20.20.20
  network 10.0.12.0 0.0.0.255 area 0
  network 10.0.65.0 0.0.0.255 area 0
  network 10.0.69.16 0.0.0.15 area 0
  network 10.0.80.0 0.0.0.255 area 0
  network 10.0.100.0 0.0.0.255 area 0
  network 192.168.25.0 0.0.0.255 area 0
  network 20.20.20.20 0.0.0.0 area 0
  network 10.2.4.4 0.0.0.3 area 0
  end

@R2
conf t
 router ospf 1
  router-id 2.2.2.2
  network 10.2.4.0 0.0.0.3 area 0
  network 10.2.4.8 0.0.0.3 area 0
  network 2.2.2.2 0.0.0.0 area 0
  end

@R3
conf t
 router ospf 1
  router-id 3.3.3.3
  network 10.2.4.4 0.0.0.3 area 0
  network 10.2.4.12 0.0.0.3 area 0
  network 3.3.3.3 0.0.0.0 area 0
  end

@R1
conf t
 router ospf 1
  router-id 1.1.1.1
  network 10.2.4.8 0.0.0.3 area 0
  network 10.2.4.12 0.0.0.3 area 0
  network 1.1.1.1 0.0.0.0 area 0
  end
 
7. Configure NAT on R1
	-create ACL
	-nat overload

@R1
conf t
 int gi0/0
  ip nat outside
 int range gi0/1-2
  ip nat inside
 ip access-list standard _myNAT
  permit 10.0.12.0 0.0.0.255
 ip nat inside source list_myNAT int gi0/0 overload
