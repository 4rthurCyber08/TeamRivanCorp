!@CME
conf t
 voice service voip
  allow-connections sip to sip                      ! Required to allow outgoing calls for SIP Traffic
  allow-connections h323 to sip
  allow-connections sip to h323
  !
  ip address trusted list
   ipv4 0.0.0.0  0.0.0.0                            ! Allow SIP traffic from this subnet
   ipv6 ::/0                                        
   exit
  sip
   silent-discard untrusted                         ! prevent SIP Scan
   header-passing                                   ! unchange SIP header info
   registrar server expires max 600 min 60        ! SIP Registrar - max registration time until register again
   bind control source-interface Gi3  ipv6-add      ! SIP Signaling interface
   bind media source-interface Gi3    ipv6-add      ! RTP Voice Streams interface
   exit
  exit
 !
 voice register global                              ! Enter SIP phone global config mode
  mode cme                                          ! Enable Cisco CME Mode
  source-address __.__.__.__  port 5060             ! SIP Address of CME
  max-dn 12                                         ! Max DN
  max-pool 12                                       ! Max SIP Phones
  authenticate register                             ! Phones must authenticate before registering
  ! no authenticate register                        ! Will not ask for authentication                        
  create profile                                    ! EQ to Create CNF-Files
  exit
 !
 voice register dn 1                                ! ---Configure DNs---
  number ____                                       ! DNumber
  name ____                                         ! Caller ID Name
  allow watch                                       ! monitor BLF
  exit
 voice register dn 2
  number ____
  name ____
  allow watch 
  exit
 !
 voice register pool 1                              ! ---Configure ePhones---
  id mac ___.___.___                                ! Phone MAC
  type ____                                         ! Phone Model
  number 1 dn 1                                     ! Button 1 > DN 1
  username ___ password ___                         ! Authentication Info
  dtmf-relay sip-notify !rtp-nte                    ! Defines how keypad tones are transmitted
  codec g711ulaw                                    ! Voice Codec (64 kbps, highest voice qualit)
  exit
 !
 voice register global
  create profile sync syncinfo.xml                  ! Update SIP phones about cnf file
  exit
 !
 dial-peer voice 1 voip
  destination-pattern 00..                          ! 00.. = Matches dialed numbers beginning with 00
  ! session protocol sipv2                          ! Set Protocol
  session target ipv4                               ! Target IP for voice
  codec g711ulaw
  dtmf-relay rtp-nte                                
  no vad                                            ! Disable Voice Activity Detection for DTMF Reliability 
  incoming called-number ___                        ! If the call comes from ___ route it based on this dial-peer

Show Commands
1. show voice register global
2. show voice register pool
3. show sip-ua register status
4. show voice call summary



SIP Flow operation
1. Phone obtains IP from DHCP
2. Phone discovers TFTP server
3. Phone downloads XML config
4. Phone sends SIP REGISTER to router
5. Router authenticates phone
6. Phone becomes available for calls
